class PM_DehackedCompatHandler : StaticEventHandler
{
    CVar cvToggle;

    private void EnsureCVars()
    {
        if (!cvToggle) cvToggle = CVar.FindCVar('projectmalice_dehacked_wolfss_replace');
    }

    override void CheckReplacement(ReplaceEvent e)
    {
        EnsureCVars();
        if (!cvToggle || !cvToggle.GetBool())
            return;

        if (e.Replacee.GetClassName() == 'WolfensteinSS')
        {
            e.Replacement = "PM_WolfSS_MP40";
        }
    }
}
