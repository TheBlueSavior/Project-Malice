
class PM_Spawner : EventHandler
{
    override void CheckReplacement(ReplaceEvent e)
    {
        if(cvar.GetCVar('pm_masterswitch').getbool()) {
            switch (e.replacee.GetClassName()) {
                case 'ZombieMan': e.Replacement="MaliceZombiemanSpawner"; break;
                case 'ShotgunGuy': e.Replacement="MaliceShotgunGuySpawner"; break;
                case 'ChaingunGuy': e.Replacement="MaliceChaingunGuySpawner"; break;
                case 'Demon': e.Replacement="MaliceDemonSpawner"; break;
                case 'Spectre': e.Replacement="MaliceSpectreSpawner"; break;
                case 'DoomImp': e.Replacement="MaliceDoomImpSpawner"; break;
                case 'LostSoul': e.Replacement="MaliceLostSoulSpawner"; break;
                case 'Revenant': e.Replacement="MaliceRevenantSpawner"; break;
                case 'Cacodemon': e.Replacement="MaliceCacodemonSpawner"; break;
                case 'PainElemental': e.Replacement="MalicePainElementalSpawner"; break;
                case 'HellKnight': e.Replacement="MaliceHellKnightSpawner"; break;
                case 'BaronOfHell': e.Replacement="MaliceBaronOfHellSpawner"; break;
                case 'Fatso': e.Replacement="MaliceFatsoSpawner"; break;
                case 'Arachnotron': e.Replacement="MaliceArachnotronSpawner"; break;
                case 'Archvile': e.Replacement="MaliceArchvileSpawner"; break;
                case 'SpiderMastermind': e.Replacement="MaliceSpiderMastermindSpawner"; break;
                case 'Cyberdemon': e.Replacement="MaliceCyberdemonSpawner"; break;
                case 'WolfensteinSS': e.Replacement="WolfensteinSS_Spawner"; break;
            }
        }
    }
}
