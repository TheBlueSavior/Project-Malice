class PM_ScoutSpawnerEH : StaticEventHandler
{
    int LevelsCleared;
    PM_SectorSpawnerThinker ThinkerRef;
    CVar cvDebug;

    private bool DebugEnabled()
    {
        if (!cvDebug) cvDebug = CVar.GetCVar('pm_scout_debug');
        return cvDebug && cvDebug.GetBool();
    }

    override void NewGame()
    {
        LevelsCleared = 0;

        if (DebugEnabled())
            Console.Printf("Scout spawner: progression reset on NewGame (LevelsCleared=%d)", LevelsCleared);
    }

    override void WorldLoaded(WorldEvent e)
    {
        if (gameState == GS_TITLELEVEL || deathmatch) return;

        if (e.IsSaveGame)
        {
            ThinkerIterator it = ThinkerIterator.Create("PM_SectorSpawnerThinker");
            ThinkerRef = PM_SectorSpawnerThinker(it.Next());
            return;
        }

        ThinkerRef = PM_SectorSpawnerThinker(new("PM_SectorSpawnerThinker"));
        if (ThinkerRef != null)
        {
            ThinkerRef.Init(LevelsCleared);

            if (DebugEnabled())
                Console.Printf("Scout spawner: progression loaded for map (LevelsCleared=%d)", LevelsCleared);

            ThinkerRef.BeginDelayedHubCheck();

            if (DebugEnabled())
                Console.Printf("Scout spawner: waiting 1 s to evaluate monster count...");
        }
    }

    override void WorldUnloaded(WorldEvent e)
    {
        if (gameState == GS_TITLELEVEL || deathmatch || e.IsReopen) return;

        LevelsCleared++;

        if (DebugEnabled())
            Console.Printf("Scout spawner: map complete → LevelsCleared now %d", LevelsCleared);
    }
}

class PM_SectorSpawnerThinker : Thinker
{

    Array<Name> ScoutActors;
    Array<int>  SpawnWeights;

    int IntervalCounter;
    int SpawnIntervalSeconds;
    int SpawnIntervalTics;
    int ResyncCounter;
    int FailCooldownTics;
    int GraceTics;

    int LevelsClearedForThisMap;

    bool PausedForHub;
    bool HubCheckPending;
    int  HubCheckDelayTics;

    int  LastIntervalSeconds;
    int  LastPrintedRemaining;

    CVar cvDebug, cvEnabled;
    CVar cvGraceSeconds, cvIntervalBase, cvIntervalStep, cvIntervalMin;
    CVar cvFailCooldown, cvClearance, cvFlatnessStep, cvFlatnessRadius;
    CVar cvAttemptSectors, cvAttemptJitter;

    private bool DebugEnabled() { return cvDebug && cvDebug.GetBool(); }

    void CacheCVars()
    {
        cvDebug          = CVar.GetCVar('pm_scout_debug');
        cvEnabled        = CVar.GetCVar('pm_scout_enabled');
        cvGraceSeconds   = CVar.GetCVar('pm_scout_grace_seconds');
        cvIntervalBase   = CVar.GetCVar('pm_scout_interval_base');
        cvIntervalStep   = CVar.GetCVar('pm_scout_interval_step');
        cvIntervalMin    = CVar.GetCVar('pm_scout_interval_min');
        cvFailCooldown   = CVar.GetCVar('pm_scout_fail_cooldown');
        cvClearance      = CVar.GetCVar('pm_scout_clearance');
        cvFlatnessStep   = CVar.GetCVar('pm_scout_flatness_step');
        cvFlatnessRadius = CVar.GetCVar('pm_scout_flatness_radius');
        cvAttemptSectors = CVar.GetCVar('pm_scout_attempt_sectors');
        cvAttemptJitter  = CVar.GetCVar('pm_scout_attempt_jitter');
    }

    void Init(int levelsCleared)
    {
        LevelsClearedForThisMap = levelsCleared;
        CacheCVars();

        ScoutActors.Clear(); SpawnWeights.Clear();
        ScoutActors.Push('PM_ScoutCharred');
        ScoutActors.Push('PM_FallenScout');
        SpawnWeights.Push(50);
        SpawnWeights.Push(50);

        LastIntervalSeconds   = -1;
        LastPrintedRemaining  = -1;

        RecomputeIntervalFromCvars();

        int gsec = 20;
        if (cvGraceSeconds) gsec = cvGraceSeconds.GetInt();
        if (gsec < 0) gsec = 0;
        GraceTics = gsec * TICRATE;

        IntervalCounter  = 0;
        ResyncCounter    = 0;
        FailCooldownTics = 0;

        PausedForHub     = true;
        HubCheckPending  = false;
        HubCheckDelayTics= 0;

        if (DebugEnabled())
            Console.Printf("Scout spawner Thinker.Init: LevelsClearedForThisMap=%d, Grace=%d s",
                           LevelsClearedForThisMap, gsec);
    }

    void BeginDelayedHubCheck()
    {

        HubCheckDelayTics = TICRATE;
        HubCheckPending   = true;
        PausedForHub      = true;
    }

    void FinalizeHubCheck()
    {
        bool shouldPause = (level.total_monsters <= 3);
        PausedForHub = shouldPause;
        HubCheckPending = false;

        if (DebugEnabled())
        {
            if (shouldPause)
                Console.Printf("Scout spawner: paused (hub/empty map detected: total_monsters=%d)", level.total_monsters);
            else
                Console.Printf("Scout spawner: active (combat map detected: total_monsters=%d)", level.total_monsters);
        }
    }

    void RecomputeIntervalFromCvars()
    {
        int baseSec = 120, stepSec = 3, minSec = 30;

        if (cvIntervalBase) baseSec = cvIntervalBase.GetInt();
        if (cvIntervalStep) stepSec = cvIntervalStep.GetInt();
        if (cvIntervalMin)  minSec  = cvIntervalMin.GetInt();

        int s = baseSec - (stepSec * LevelsClearedForThisMap);
        if (s < minSec) s = minSec;

        SpawnIntervalSeconds = s;
        SpawnIntervalTics    = SpawnIntervalSeconds * TICRATE;

        if (LastIntervalSeconds != SpawnIntervalSeconds)
        {
            LastIntervalSeconds = SpawnIntervalSeconds;
            if (DebugEnabled())
                Console.Printf("Scout spawner: interval set → base=%d, step=%d, min=%d, clears=%d ⇒ %d s",
                               baseSec, stepSec, minSec, LevelsClearedForThisMap, SpawnIntervalSeconds);
        }
    }

    override void Tick()
    {

        if (cvEnabled && !cvEnabled.GetBool()) return;
        if (SpawnIntervalTics <= 0) return;

        if (HubCheckPending)
        {
            if (HubCheckDelayTics > 0)
            {
                HubCheckDelayTics--;
                Super.Tick();
                return;
            }
            FinalizeHubCheck();

        }

        if (PausedForHub)
        {
            Super.Tick();
            return;
        }

        ResyncCounter++;
        if (ResyncCounter >= TICRATE)
        {
            ResyncCounter = 0;
            RecomputeIntervalFromCvars();

            if (DebugEnabled())
            {
                int remaining = (SpawnIntervalTics - IntervalCounter) / TICRATE;
                if (remaining < 0) remaining = 0;

                bool shouldPrint = false;
                if (remaining <= 10)
                    shouldPrint = (remaining != LastPrintedRemaining);
                else if ((remaining % 5) == 0)
                    shouldPrint = (remaining != LastPrintedRemaining);

                if (shouldPrint)
                {
                    LastPrintedRemaining = remaining;
                    Console.Printf("Scout spawner: next attempt in %d s (clears=%d)", remaining, LevelsClearedForThisMap);
                }
            }
        }

        if (GraceTics > 0)
        {
            GraceTics--;
            Super.Tick();
            return;
        }

        if (FailCooldownTics > 0)
        {
            FailCooldownTics--;
            Super.Tick();
            return;
        }

        IntervalCounter++;

        if (IntervalCounter >= SpawnIntervalTics)
        {
            if (DebugEnabled())
                Console.Printf("Scout spawner: interval elapsed → attempting spawn (clears=%d)", LevelsClearedForThisMap);

            bool spawned = TrySpawnOneRandomActor();
            if (spawned)
            {
                IntervalCounter = 0;

                if (DebugEnabled())
                    Console.Printf("Scout spawner: spawned (clears=%d, interval=%d s)",
                                   LevelsClearedForThisMap, SpawnIntervalSeconds);

                LastPrintedRemaining = -1;
            }
            else
            {
                int coolSec = 0;
                if (cvFailCooldown) coolSec = cvFailCooldown.GetInt();
                if (coolSec < 0) coolSec = 0;
                FailCooldownTics = coolSec * TICRATE;

                if (DebugEnabled())
                    Console.Printf("Scout spawner: spawn failed, cooldown %d s (clears=%d)", coolSec, LevelsClearedForThisMap);

                LastPrintedRemaining = -1;
            }
        }

        Super.Tick();
    }

    bool PrecheckSpot(Sector sec, Vector2 p2, out double outFloorZ)
    {
        int clearance = 56, flatStep = 24, flatRadius = 16;
        if (cvClearance)      clearance  = cvClearance.GetInt();
        if (cvFlatnessStep)   flatStep   = cvFlatnessStep.GetInt();
        if (cvFlatnessRadius) flatRadius = cvFlatnessRadius.GetInt();

        double floorZ  = sec.floorplane.ZAtPoint(p2);
        double ceilZ   = sec.ceilingplane.ZAtPoint(p2);
        if (ceilZ - floorZ < clearance) return false;

        Vector2 dirs[8];
        dirs[0].x=-1; dirs[0].y= 0; dirs[1].x= 1; dirs[1].y= 0;
        dirs[2].x= 0; dirs[2].y=-1; dirs[3].x= 0; dirs[3].y= 1;
        dirs[4].x=-1; dirs[4].y=-1; dirs[5].x= 1; dirs[5].y=-1;
        dirs[6].x=-1; dirs[6].y= 1; dirs[7].x= 1; dirs[7].y= 1;

        for (int i = 0; i < 8; i++)
        {
            Vector2 pn;
            pn.x = p2.x + dirs[i].x * flatRadius;
            pn.y = p2.y + dirs[i].y * flatRadius;

            Vector3 probe;
            probe.x = pn.x; probe.y = pn.y; probe.z = floorZ;
            if (!level.IsPointInLevel(probe)) return false;

            double nf = sec.floorplane.ZAtPoint(pn);
            if (abs(nf - floorZ) > flatStep) return false;
        }

        outFloorZ = floorZ;
        return true;
    }

    bool TrySpawnOneRandomActor()
    {
        if (ScoutActors.Size() == 0 || level.sectors.Size() == 0)
            return false;

        int idx = GetRandomActorIndex();
        Name typeName = ScoutActors[idx];

        class<Actor> cls = typeName;
        if (cls == null)
        {
            if (DebugEnabled())
                Console.Printf("Scout spawner: unknown class '%s'", typeName);
            return false;
        }

        int SectorTries = 128, JitterTries = 8;
        int JitterMax   = 128;

        if (cvAttemptSectors) SectorTries = cvAttemptSectors.GetInt();
        if (cvAttemptJitter)  JitterTries = cvAttemptJitter.GetInt();

        if (SectorTries < 8)   SectorTries = 8;
        if (SectorTries > 256) SectorTries = 256;
        if (JitterTries < 1)   JitterTries = 1;
        if (JitterTries > 16)  JitterTries = 16;

        for (int i = 0; i < SectorTries; i++)
        {
            int sidx = Random(0, level.sectors.Size() - 1);
            Sector sec = level.sectors[sidx];

            Vector2 c = sec.centerspot;

            for (int j = 0; j < JitterTries; j++)
            {
                double ox = c.x + Random(-JitterMax, JitterMax);
                double oy = c.y + Random(-JitterMax, JitterMax);
                Vector2 p2; p2.x = ox; p2.y = oy;

                Sector actualSec = level.PointInSector(p2);
                if (actualSec == null) continue;

                double floorZ;
                if (!PrecheckSpot(actualSec, p2, floorZ)) continue;

                Vector3 pos;
                pos.x = p2.x; pos.y = p2.y; pos.z = floorZ;

                if (!level.IsPointInLevel(pos)) continue;

                let mo = Actor.Spawn(cls, pos);
                if (mo && mo.TestMobjLocation())
                {
                    if (DebugEnabled())
                        Console.Printf("A scout is investigating...");
                    return true;
                }
                else if (mo)
                {
                    mo.Destroy();
                }
            }
        }
        return false;
    }

    int GetRandomActorIndex()
    {
        int total = 0;
        for (int i = 0; i < SpawnWeights.Size(); i++)
            total += SpawnWeights[i];
        if (total <= 0) return 0;

        int roll = Random(0, total - 1);
        for (int i = 0; i < SpawnWeights.Size(); i++)
        {
            roll -= SpawnWeights[i];
            if (roll < 0) return i;
        }
        return 0;
    }
}
