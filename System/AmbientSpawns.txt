class PM_FodderMapStamp : Actor
{
    Default
    {
        +NOINTERACTION
        +NOBLOCKMAP
        +NOSECTOR
        +INVISIBLE
        RenderStyle "None";
        Alpha 0.0;
        Scale 0;
    }
}

class PM_FodderDropper : StaticEventHandler
{
    bool   doSpawn;
    int    spawnDelay;
    double roll;

    string lastMapName;

    CVar cvDebug, cvMode, cvProg, cvIncrement, cvDivider;

    private void EnsureCVars()
    {
        if (!cvDebug)     cvDebug     = CVar.FindCVar('projectmalice_fodder_debug');
        if (!cvMode)      cvMode      = CVar.FindCVar('projectmalice_foddermode');
        if (!cvProg)      cvProg      = CVar.FindCVar('projectmalice_fodderprogression');
        if (!cvIncrement) cvIncrement = CVar.FindCVar('projectmalice_fodderincrement');
        if (!cvDivider)   cvDivider   = CVar.FindCVar('projectmalice_fodderdivider');
    }

    private bool DebugEnabled()
    {
        EnsureCVars();
        return cvDebug && cvDebug.GetBool();
    }

    override void NewGame()
    {
        EnsureCVars();
        if (cvProg) cvProg.SetFloat(0.0);

        lastMapName = "";

        if (DebugEnabled())
            Console.Printf("[PM:Fodder] New Game: progression reset to 0.0");
    }

    override void WorldLoaded(WorldEvent e)
    {
        if (gamestate == GS_TITLELEVEL || deathmatch) return;
        EnsureCVars();

        int mode = cvMode ? cvMode.GetInt() : 0;
        if (mode == 1)
        {
            if (lastMapName == "")
            {
                if (DebugEnabled())
                    Console.Printf("[PM:Fodder] First map this run: progression stays at %.2f",
                        cvProg ? cvProg.GetFloat() : 0.0);
            }
            else if (level.MapName != lastMapName && !e.IsSaveGame)
            {
                if (cvProg && cvIncrement)
                {
                    double newVal = cvProg.GetFloat() + cvIncrement.GetFloat();
                    if (newVal > 20.0) newVal = 20.0;
                    cvProg.SetFloat(newVal);

                    if (DebugEnabled())
                    {
                        Console.Printf("[PM:Fodder] Map change %s -> %s: +%.2f -> %.2f (cap 20.0)",
                            lastMapName, level.MapName, cvIncrement.GetFloat(), newVal);
                    }
                }
            }
            else
            {

                if (DebugEnabled())
                    Console.Printf("[PM:Fodder] Restart detected on %s: progression unchanged.",
                        level.MapName);
            }
        }

        lastMapName = level.MapName;

        bool alreadyStamped = MapHasFodderStamp();

        doSpawn    = !alreadyStamped;
        spawnDelay = 1;
        roll       = 0.0;

        if (DebugEnabled())
        {
            if (mode == 1)
            {
                Console.Printf("[PM:Fodder] Mode: Progression. Ceiling this map: %.2f (%s)",
                    cvProg ? cvProg.GetFloat() : 0.0,
                    alreadyStamped ? "reload detected: no initial drop" : "fresh entry");
            }
            else
            {
                Console.Printf("[PM:Fodder] Mode: Static. Divider: %.2f (%s)",
                    cvDivider ? cvDivider.GetFloat() : 0.0,
                    alreadyStamped ? "reload detected: no initial drop" : "fresh entry");
            }
        }
    }

    override void WorldTick()
    {
        if (!doSpawn) return;

        if (spawnDelay-- > 0) return;
        doSpawn = false;

        EnsureCVars();

        Array<double> dropX, dropY, dropZ;

        Actor mo;
        let it = ThinkerIterator.Create("Actor");
        while (mo = Actor(it.Next()))
        {
            if (mo is "PlayerPawn" || !mo.bISMONSTER) continue;
            if (mo.bFLOAT) continue;
            dropX.Push(mo.pos.X);
            dropY.Push(mo.pos.Y);
            dropZ.Push(mo.pos.Z);
        }
        if (dropX.Size() == 0) return;

        int mode = cvMode ? cvMode.GetInt() : 0;
        int extraCount;

        if (mode == 0)
        {
            double div = cvDivider ? cvDivider.GetFloat() : 0.0;
            roll = frandom(0.0, div);
            extraCount = int(level.total_monsters * roll);
        }
        else
        {
            double ceiling = cvProg ? cvProg.GetFloat() : 0.0;
            roll = frandom(0.0, ceiling);
            extraCount = int(level.total_monsters * roll);
        }

        if (DebugEnabled())
        {
            Console.Printf("[PM:Fodder] %s: Mode=%d Progression=%.2f Roll=%.2f Extra=%d",
                level.MapName, mode,
                cvProg ? cvProg.GetFloat() : 0.0,
                roll, extraCount);
        }

        for (int i = 0; i < extraCount; i++)
        {
            int idx = random(0, dropX.Size() - 1);
            Vector3 sp = (dropX[idx], dropY[idx], dropZ[idx]);
            let a = Actor.Spawn("PM_HorrorPassiveSpawn", sp);
            if (a) a.angle = random(0, 7) * 45;
        }

        SpawnOrEnsureStamp();
    }

    bool MapHasFodderStamp()
    {
        let it = ThinkerIterator.Create("PM_FodderMapStamp");
        return it.Next() != null;
    }

    void SpawnOrEnsureStamp()
    {
        if (MapHasFodderStamp()) return;

        for (int i = 0; i < MAXPLAYERS; i++)
        {
            if (playeringame[i] && players[i].mo)
            {
                Actor.Spawn("PM_FodderMapStamp", players[i].mo.pos);
                return;
            }
        }
    }
}
