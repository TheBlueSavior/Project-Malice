class PM_ColdBreathToken : Inventory
{
	int breathTimer;
	int puffCount;
	int nextPuff;

	Default
	{
		Inventory.MaxAmount 1;
		+INVENTORY.UNDROPPABLE;
		+INVENTORY.ALWAYSPICKUP;
		+INVENTORY.AUTOACTIVATE;
	}

	override void DoEffect()
	{
		Super.DoEffect();
		if (!Owner || Owner.health <= 0 || !Owner.player) { Destroy(); return; }

		breathTimer++;

		if (breathTimer > 430) { Destroy(); return; }

		int billowDuration = 12;
		int billowStart1 = 10;
		int billowStart2 = billowStart1 + 70;
		int billowStart3 = billowStart2 + 70;

		bool inBillow = (breathTimer >= billowStart1 && breathTimer < billowStart1 + billowDuration)
					 || (breathTimer >= billowStart2 && breathTimer < billowStart2 + billowDuration)
					 || (breathTimer >= billowStart3 && breathTimer < billowStart3 + billowDuration);

		if (!inBillow) return;

		double mouthZ = Owner.Height * 0.55;
		double fwdDist = 10.0;
		double fwdX = cos(Owner.angle) * fwdDist;
		double fwdY = sin(Owner.angle) * fwdDist;

		for (int i = 0; i < random(2, 3); i++)
		{
			Owner.A_SpawnParticleEx("",
				TexMan.CheckForTexture("GNSMI0"),
				style: STYLE_Normal,
				flags: SPF_ROLL,
				lifetime: random(110, 210),
				size: frandom(2, 5),
				xoff: fwdX + frandom(-3, 3),
				yoff: fwdY + frandom(-3, 3),
				zoff: mouthZ + frandom(-2, 2),
				velx: cos(Owner.angle) * frandom(0.3, 0.8) + frandom(-0.1, 0.1),
				vely: sin(Owner.angle) * frandom(0.3, 0.8) + frandom(-0.1, 0.1),
				velz: frandom(0.1, 0.4),
				startalphaf: frandom(0.10, 0.25),
				fadestepf: -1,
				sizestep: frandom(0.5, 1.25),
				startroll: random(-180, 180),
				rollvel: frandom(-1.5, 1.5)
			);
		}

		if (breathTimer % 2 == 0)
		{
			Owner.A_SpawnParticleEx("",
				TexMan.CheckForTexture("GNSMI0"),
				style: STYLE_Normal,
				flags: SPF_ROLL,
				lifetime: random(140, 260),
				size: frandom(3, 6),
				xoff: fwdX + frandom(-4, 4),
				yoff: fwdY + frandom(-4, 4),
				zoff: mouthZ + frandom(-2, 2),
				velx: cos(Owner.angle) * frandom(0.2, 0.5) + frandom(-0.15, 0.15),
				vely: sin(Owner.angle) * frandom(0.2, 0.5) + frandom(-0.15, 0.15),
				velz: frandom(0.15, 0.5),
				startalphaf: frandom(0.05, 0.14),
				fadestepf: -1,
				sizestep: frandom(0.75, 1.5),
				startroll: random(-180, 180),
				rollvel: frandom(-2, 2)
			);
		}
	}

	States
	{
	Spawn:
		TNT1 A -1;
		Stop;
	}
}

	{

	void PMGhost_BeginHaunt()
	{
		bNoClip     = true;
		bShootable  = false;
		bThruActors = true;
		A_SetRenderStyle(0.0,STYLE_TranslucentStencil);
	}

	void PMGhost_BeginAttack()
	{
		PMG_InAttack   = true;
		bShootable     = true;
		A_SetRenderStyle(1.0,STYLE_TranslucentStencil);
		A_RadiusGive("PM_GhostAttackStim", PMG_SignalRadius, RGF_MONSTERS, 1);
		PMGhost_Debug("BeginAttack + Rally");
	}

	void PMGhost_EndAttack()
	{
		PMG_InAttack = false;
		bShootable   = false;
		A_SetRenderStyle(0.0,STYLE_TranslucentStencil);

		let stim = FindInventory("PM_GhostAttackStim");
		if (stim) stim.Destroy();

		PMGhost_Debug("EndAttack");
	}

	bool PMGhost_AcquireAnyPlayer()
	{
		for (int i = 0; i < MAXPLAYERS; i++)
		{
			if (!playeringame[i]) continue;
			PlayerInfo pi = players[i];
			PlayerPawn pp = pi.mo;
			if (pp != null && pp.health > 0 && pi.playerstate != PST_DEAD)
			{
				target = pp;
				return true;
			}
		}
		return false;
	}

		TNT1 A 0 A_JumpIf(target == null || (target.player && target.player.playerstate == PST_DEAD), "Spawn");

		TNT1 A 0
		{
			let stim = FindInventory("PM_GhostAttackStim");
			if (stim) { stim.Destroy(); SetStateLabel("BeginAttack"); }
		}

		TNT1 A 0
		{
			if (Random(0,31) < 8)

				int r = Random(0, 2);

				{
					A_SpawnItemEx("PM_GhostWhisperer",random(-128,128),random(-128,128),0,0,0,0);
					PMGhost_Debug("HauntAction: Whisper");
				}
				else if (r == 1)
				{
					SetStateLabel("HauntManifest");
				}
				else
				{
					A_RadiusGive("PM_ColdBreathToken", 576, RGF_PLAYERS, 1);
					PMGhost_Debug("HauntAction: ColdBreath");
				}
			}
		}
		TNT1 A 0 A_Jump(128, "HauntWander");
		TNT1 AAAAAAAAAAAAAAAAAAAAAAAAAAAAAA 3;
		TNT1 A 0 { bNOGRAVITY = false; }
		Goto HauntLoop;

	HauntWander:
		TNT1 AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA 2 A_Wander();                  
		TNT1 A 0 A_JumpIfTargetInLOS("HauntLoop");
		TNT1 AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA 1 A_Chase(null, null);
		Goto HauntLoop;

		TNT1 A 0 A_JumpIfCloser(52,"Bite");
		SAR2 A 7 A_FaceTarget();
		SAR2 E 4 A_FaceTarget();
		TNT1 A 0 ThrustThingZ(0,20,0,1);
		TNT1 A 0 A_Recoil(-7);

		TNT1 A 0 A_SpawnProjectile("PM_MeleeHitbox",5,0,0,2);

		SAR2 F 8 A_FaceTarget();
