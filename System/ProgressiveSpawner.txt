class PM_ProgressiveSpawnerHandler : StaticEventHandler
{

    string lastMapName;

    CVar cvMode, cvProg, cvFullUnlock, cvDebug;

    Array<Name>   tC;
    Array<int>    tW;
    Array<int>    tT;
    Array<double> tEW;

    private void EnsureCVars()
    {
        if (!cvMode)       cvMode       = CVar.FindCVar('pm_spawner_mode');
        if (!cvProg)       cvProg       = CVar.FindCVar('pm_spawner_progression');
        if (!cvFullUnlock) cvFullUnlock = CVar.FindCVar('pm_spawner_fullunlock');
        if (!cvDebug)      cvDebug      = CVar.FindCVar('pm_spawner_debug');
    }

    private bool DebugEnabled()
    {
        EnsureCVars();
        return cvDebug && cvDebug.GetBool();
    }

    override void NewGame()
    {
        EnsureCVars();
        if (cvProg) cvProg.SetInt(0);
        lastMapName = "";

        if (DebugEnabled())
            Console.Printf("[PM:Spawner] New Game: progression reset to 0");
    }

    override void WorldLoaded(WorldEvent e)
    {
        if (gamestate == GS_TITLELEVEL || deathmatch) return;
        EnsureCVars();

        bool haveLast   = (lastMapName != "");
        bool mapChanged = haveLast && (level.MapName != lastMapName);

        if (mapChanged && !e.IsSaveGame && cvProg)
        {
            int newVal = cvProg.GetInt() + 1;
            cvProg.SetInt(newVal);

            if (DebugEnabled())
            {
                int full = cvFullUnlock ? cvFullUnlock.GetInt() : 25;
                Console.Printf("[PM:Spawner] Map change %s -> %s: progression now %d/%d",
                    lastMapName, level.MapName, newVal, full);
            }
        }
        else if (!haveLast && DebugEnabled())
        {
            int cur  = cvProg ? cvProg.GetInt() : 0;
            int full = cvFullUnlock ? cvFullUnlock.GetInt() : 25;
            Console.Printf("[PM:Spawner] First map this run: %s, progression=%d/%d",
                level.MapName, cur, full);
        }

        lastMapName = level.MapName;
    }

    override void CheckReplacement(ReplaceEvent e)
    {
        EnsureCVars();
        if (!cvMode || cvMode.GetInt() != 1) return;

        Name cls = e.Replacee.GetClassName();
        if (!BuildTable(cls)) return;

        Name picked = PickProgressive();
        if (picked != 'None')
        {
            e.Replacement = picked;

            if (DebugEnabled())
            {
                int cur = cvProg ? cvProg.GetInt() : 0;
                Console.Printf("[PM:Spawner] %s -> %s (prog=%d)", cls, picked, cur);
            }
        }
    }

    double CalcWeight(int baseWeight, int tier, int maxTier, int maxBW, double progress)
    {

        if (tier == 0)
        {
            return double(baseWeight) * (1.0 - progress);
        }

        double unlockFrac = double(tier) / (maxTier + 1);
        if (progress < unlockFrac) return 0.0;

        double rampRange = 1.0 - unlockFrac;
        if (rampRange <= 0.0) return double(baseWeight);

        double maturity = (progress - unlockFrac) / rampRange;
        double invertedW = double(maxBW + 1 - baseWeight);
        return invertedW * maturity;
    }

    Name PickProgressive()
    {
        if (tC.Size() == 0) return 'None';

        int full = cvFullUnlock ? cvFullUnlock.GetInt() : 25;
        if (full < 1) full = 1;
        int cur = cvProg ? cvProg.GetInt() : 0;
        double progress = double(cur) / full;
        if (progress > 1.0) progress = 1.0;

        int maxTier = 0;
        int maxBW   = 0;
        for (int i = 0; i < tT.Size(); i++)
        {
            if (tT[i] > maxTier) maxTier = tT[i];
            if (tW[i] > maxBW)   maxBW   = tW[i];
        }
        if (maxTier < 1) maxTier = 1;

        tEW.Clear();
        double totalWeight = 0;
        for (int i = 0; i < tC.Size(); i++)
        {
            double ew = CalcWeight(tW[i], tT[i], maxTier, maxBW, progress);
            tEW.Push(ew);
            totalWeight += ew;
        }

        if (totalWeight <= 0.0) return tC[0];

        double roll = frandom(0.0, totalWeight);
        for (int i = 0; i < tEW.Size(); i++)
        {
            roll -= tEW[i];
            if (roll <= 0.0) return tC[i];
        }
        return tC[tC.Size() - 1];
    }

    private void E(Name cls, int weight, int tier)
    {
        tC.Push(cls);
        tW.Push(weight);
        tT.Push(tier);
    }

    bool BuildTable(Name cls)
    {
        tC.Clear(); tW.Clear(); tT.Clear();

        if      (cls == 'ZombieMan')        BuildZombiemanTable();
        else if (cls == 'ShotgunGuy')       BuildShotgunGuyTable();
        else if (cls == 'ChaingunGuy')      BuildChaingunGuyTable();
        else if (cls == 'DoomImp')          BuildDoomImpTable();
        else if (cls == 'Demon')            BuildDemonTable();
        else if (cls == 'Spectre')          BuildSpectreTable();
        else if (cls == 'LostSoul')         BuildLostSoulTable();
        else if (cls == 'Revenant')         BuildRevenantTable();
        else if (cls == 'Cacodemon')        BuildCacodemonTable();
        else if (cls == 'PainElemental')    BuildPainElementalTable();
        else if (cls == 'HellKnight')       BuildHellKnightTable();
        else if (cls == 'BaronOfHell')      BuildBaronOfHellTable();
        else if (cls == 'Fatso')            BuildFatsoTable();
        else if (cls == 'Arachnotron')      BuildArachnotronTable();
        else if (cls == 'Archvile')         BuildArchvileTable();
        else if (cls == 'SpiderMastermind') BuildSpiderMastermindTable();
        else if (cls == 'Cyberdemon')       BuildCyberdemonTable();
        else return false;

        return tC.Size() > 0;
    }
